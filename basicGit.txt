### clone repository (only need to do this once) ###
	git clone https://github.com/Jeff-ChenFan-Wang/StatAnalysis.git



### Adding your changes to git ###
	git add -A
	git commit -m "i added something ...." 
	git push


### getting the newest updates ###
	git pull


### create new branch ###
	git checkout -b (branchName)
	git add -A
	git commit -m "comment what you added"
	git push --set-upstream origin (branchName)
